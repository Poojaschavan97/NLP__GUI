from html.parser import commentclose
from tkinter import *
from tkinter.messagebox import showerror

from myDB import Database
from tkinter import messagebox
from myapi import API

class NLPAPP:

    def __init__(self):

        #create DB object
        self.dbo=Database() #this can now access any method from database class

        #fetching api
        self.apio=API()

        #load login GUI
        self.root=Tk()  # STORING OBJECTS OF TK
        self.root.title('NLPAPP')  # title in head space
        self.root.iconbitmap('RESOURCES/favicon.ico')  # for adding favicon
        self.root.geometry('500x500')  # Size of the gui
        self.root.configure(bg='#c1dfd1')# background color, font style
        self.login_gui() # object of gui
        self.root.mainloop()  # to see GUI

    def login_gui(self):
        self.clear()

        heading=Label(self.root, text='NLPApp') # adding text
        heading.pack(pady=(30,20)) # to see text on gui,pady for margin around text
        heading.configure(font=('Castellar',24,'bold'), bg='#c1dfd1',fg='purple')# font color and background color also can be added in Label

        label1=Label(self.root,text='enter Email')
        label1.pack(pady=(10,10))

        self.email_input=Entry(self.root,width=50) #for user input,self is added cause this var is further used
        self.email_input.pack(pady=(5,10),ipady=3) # height cant be entered in entry so ipady is used y for vertical

        label2 = Label(self.root, text='enter Password')
        label2.pack(pady=(10, 10))

        self.password_input = Entry(self.root, width=50,show='*')  # for user input,self is added cause this var is further used
        self.password_input.pack(pady=(5, 10), ipady=3)# show is used to hide password

        login_button=Button(self.root,text='Login',width=10,height=1, command=self.perform_login)
        login_button.pack(pady=(10,10))

        label3=Label(self.root,text='Not a member?')
        label3.pack(pady=(20,10))

        redirect_button = Button(self.root, text='Register Now', width=10, height=1,command=self.register_gui)
        # command is used to direct a function to a button
        redirect_button.pack(pady=(10, 10))

    def register_gui(self):
        self.clear()

        heading = Label(self.root, text='NLPApp')  # adding text
        heading.pack(pady=(30, 20))  # to see text on gui,pady for margin around text
        heading.configure(font=('Castellar', 24, 'bold'), bg='#c1dfd1',
                          fg='purple')  # font color and background color also can be added in Label

        label0= Label(self.root, text='enter Name')
        label0.pack(pady=(10, 10))

        self.name_input = Entry(self.root, width=50)  # for user input,self is added cause this var is further used
        self.name_input.pack(pady=(5, 10), ipady=3)

        label1 = Label(self.root, text='enter Email')
        label1.pack(pady=(10, 10))

        self.email_input = Entry(self.root, width=50)  # for user input,self is added cause this var is further used
        self.email_input.pack(pady=(5, 10), ipady=3)  # height cant be entered in entry so ipady is used y for vertical

        label2 = Label(self.root, text='enter Password')
        label2.pack(pady=(10, 10))

        self.password_input = Entry(self.root, width=50,
                                    show='*')  # for user input,self is added cause this var is further used
        self.password_input.pack(pady=(5, 10), ipady=3)  # show is used to hide password

        register_button = Button(self.root, text='register', width=10, height=1, command=self.perform_registration)
        register_button.pack(pady=(10, 10))

        label3 = Label(self.root, text='Already a member?')
        label3.pack(pady=(20, 10))

        redirect_button = Button(self.root, text='Login Now', width=10, height=1, command=self.login_gui)
        # command is used to direct a function to a button
        redirect_button.pack(pady=(10, 10))

    def clear(self):
        # clear the existing gui
        for i in self.root.pack_slaves():
            i.destroy()

    def perform_registration(self):
        #fetch data  from gui
        name=self.name_input.get()
        email=self.email_input.get()
        password=self.password_input.get()

        response=self.dbo.add_data(name,email,password)

        if response:
            messagebox.showinfo('success','reg successful,you can login now')
        else:
            messagebox.showerror('error','email id already exists')

    def perform_login(self):
        email=self.email_input.get()
        password=self.password_input.get()

        response=self.dbo.search(email,password)
        if response:
            messagebox.showinfo('success','login successful')
            self.home_gui()
        else:
            messagebox,showerror('error','incorrect email/password')
    def home_gui(self):
        self.clear()

        heading = Label(self.root, text='NLPApp')  # adding text
        heading.pack(pady=(30, 20))  # to see text on gui,pady for margin around text
        heading.configure(font=('Castellar', 24, 'bold'), bg='#c1dfd1',
                          fg='purple')

        sentiment_btn=Button(self.root, text='sentiment Analysis',width=50,height=4, command=self.sentiment_gui)
        sentiment_btn.pack(pady=(10,10))

        ner_btn = Button(self.root, text='named entity registration', width=50, height=4, command=self.ner_gui)
        ner_btn.pack(pady=(10, 10))

        summary_btn = Button(self.root, text='Summary Creator', width=50, height=4, command=self.summary_gui)
        summary_btn.pack(pady=(10, 10))

        logout_btn.pack(pady=(10, 10))

    def sentiment_gui(self):
        self.clear()

        heading = Label(self.root, text='NLPApp')  # adding text
        heading.pack(pady=(20, 10))  # to see text on gui,pady for margin around text
        heading.configure(font=('Castellar', 24, 'bold'), bg='#c1dfd1',
                          fg='purple')

        heading1 = Label(self.root, text='Sentiment Analysis')  # adding text
        heading1.pack(pady=(10, 10))  # to see text on gui,pady for margin around text
        heading1.configure(font=('Forte', 18, 'bold'), bg='#c1dfd1',
                          fg='coral')

        label1 = Label(self.root, text='Enter text')
        label1.pack(pady=(20, 10))
        self.sentiment_input = Entry(self.root, width=50)
        self.sentiment_input.pack(pady=(5, 10), ipady=3)

        sentiment_btn = Button(self.root, text='analyze sentiment', width=20, height=1, command=self.do_sentiment_analysis)
        sentiment_btn.pack(pady=(10, 10))

        self.response= Label(self.root, text='',bg='#c1dfd1')
        self.response.pack(pady=(20, 10))
        self.response.configure(font=('Arial Narrow',16))

        back_btn = Button(self.root, text='Back', width=10, height=1, command=self.home_gui)
        back_btn.pack(pady=(10, 10))

    def do_sentiment_analysis(self):
        text=self.sentiment_input.get()
        result=self.apio.sentiment_analysis(text)

        txt=''
        for i in result['scored_labels']:
            txt=txt+i['label']+' --> '+str(i['score'])+'\n'

        self.response['text']=txt

    def do_ner(self):
        ner_text=self.ner_input.get()
        entity_search=self.entity.get()
        result=self.apio.named_entity_recognization(ner_text,entity_search)

        txt=''
        for i in result['entities']:
            txt=txt+i['text']+'\n'

        self.response['text']=txt
